% -- spacing funcs
\newcommand\hfEnumReduceSpacing{
  \RequirePackage{enumitem}
  \setlength{\itemsep}{0pt}
}
\newcommand\hfParagraphReduceSpacing{
  \linespread{1}
  \setlength{\parskip}{0pt}
  \setlength{\parsep}{0pt}
}
\newcommand\hfCaptionReduceSpacing{
  \RequirePackage{caption}
  \setlength{\abovecaptionskip}{1pt}
  \setlength{\belowcaptionskip}{1pt}
}
\newcommand\hfTitleReduceSpacing{
  \RequirePackage[compact]{titlesec}
  \titlespacing{\section}{0pt}{*0}{*0}
  \titlespacing{\subsection}{0pt}{*0}{*0}
  \titlespacing{\subsubsection}{0pt}{*0}{*0}
}
\newcommand\hfReduceSpacing{
  \hfEnumReduceSpacing
  \hfCaptionReduceSpacing
  \hfParagraphReduceSpacing
  \hfTitleReduceSpacing
}
